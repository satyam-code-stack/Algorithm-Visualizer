// C++ program for implementation of Bubble Sort
    #include 
        using namespace std;
        
        void bubbleSort(int arr[], int n) {
            for (int i = 0; i < n - 1; i++) {
                // Flag to check if any swapping happened
                bool swapped = false;
        
                for (int j = 0; j < n - i - 1; j++) {
                    if (arr[j] > arr[j + 1]) {
                        // Swap if the element is greater than the next element
                        swap(arr[j], arr[j + 1]);
                        swapped = true;
                    }
                }
        
                // If no swapping happened in this pass, the array is already sorted
                if (!swapped)
                    break;
            }
        }
        
        void printArray(int arr[], int n) {
            for (int i = 0; i < n; i++) {
                cout << arr[i] << " ";
            }
            cout << endl;
        }
        
        int main() {
            int arr[] = {64, 34, 25, 12, 22, 11, 90};
            int n = sizeof(arr) / sizeof(arr[0]);
        
            cout << "Original array: ";
            printArray(arr, n);
        
            bubbleSort(arr, n);
        
            cout << "Sorted array: ";
            printArray(arr, n);
        
            return 0;
        }
